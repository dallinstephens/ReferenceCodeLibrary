Primary Code Tidbit 
<div ng-app="myGroceryList" ng-controller="controllerName">
  <ul>
    <li ng-repeat="x in foods">{{x}}<span ng-click="removeItem($index)">&nbsp;x</span></li>
  </ul>

  <div>
    <input ng-model="addFood" value="Type food you want to add">
    <button ng-click="addItem()" onclick="resultBoxAreaAdjust()">Add Food</button>
  </div>

  <p>Click the small red x to remove an item from the grocery list.</p>
</div>

<script>
  var app = angular.module('myGroceryList', []);
  app.controller('controllerName',
  function($scope) {
    $scope.foods=["Egg Whites", "Greek Yogurt", "Clementites", "Boneless Chicken Breast"];
    $scope.addItem = function () {
      $scope.foods.push($scope.addFood);
    }
    $scope.removeItem = function (x) {
      $scope.foods.splice(x, 1);
    }
  });
</script>
